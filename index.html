import random from pyodide.ffi import create_proxy # თქვენი ორიგინალური ლოგიკა options = ['rock', 'paper', 'scissors'] def play_game(event): # ვიგებთ რომელ ღილაკს დააჭირეს ID-ს მიხედვით btn_id = event.target.id user_choice = btn_id.replace('-btn', '') computer_choice = random.choice(options) result = "" if user_choice == computer_choice: result = "ფრეა! 🤝" elif (user_choice == 'rock' and computer_choice == 'scissors') or \ (user_choice == 'paper' and computer_choice == 'rock') or \ (user_choice == 'scissors' and computer_choice == 'paper'): result = "თქვენ მოიგეთ! 🎉" else: result = "თქვენ წააგეთ! 🤖" output_div = Element("output") output_div.write(f"თქვენ: {user_choice.upper()}\nკომპიუტერი: {computer_choice.upper()}\nშედეგი: {result}") # ღილაკებზე ფუნქციების მიბმა rock_proxy = create_proxy(play_game) Element("rock-btn").element.addEventListener("click", rock_proxy) paper_proxy = create_proxy(play_game) Element("paper-btn").element.addEventListener("click", paper_proxy) scissors_proxy = create_proxy(play_game) Element("scissors-btn").element.addEventListener("click", scissors_proxy) Element("output").write("თამაში მზად არის! აირჩიეთ.")
